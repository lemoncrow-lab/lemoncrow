## Success Criteria

1. `from atelier.core.capabilities.prompt_compilation import PromptBlock, Stability, BlockKind` works
2. `PromptBlock` construction enforces all invariants (empty content, id format, override reason, cacheable coercion)
3. `version_hash` is deterministic: same content → same sha256 across processes
4. `token_estimate` uses tiktoken when available and falls back gracefully
5. All 8 unit tests pass: `uv run pytest tests/core/capabilities/prompt_compilation -q`
6. `make lint && make typecheck` pass

---

## Threat Model

| Threat | Mitigation |
|--------|-----------|
| Mutable PromptBlock allows silent cache poisoning | `frozen=True` on dataclass; cached hash validates at access |
| `id` with spaces or special chars breaks cache keys and trace IDs | Enforce `^[a-z0-9_./:-]+$` regex at `__post_init__` |
| Caller overrides stability without documenting why | Require `stability_override_reason` when stability ≠ default |
| VOLATILE/TURN blocks incorrectly marked cacheable | Force `cacheable=False` in `__post_init__` regardless of caller intent |
| tiktoken missing in some environments | `max(1, len(text) // 4)` fallback so token_estimate never raises |

**Task 08-02**: Implement `models.py`

File: `src/atelier/core/capabilities/prompt_compilation/models.py`

Full implementation:
```python
"""Block model for the Atelier prompt compiler.

PromptBlock, Stability, BlockKind, and DEFAULT_STABILITY are the foundational
types that P1 (compiler), P2 (linter), P3 (providers), and P5 (trace) all
depend on.
"""
from __future__ import annotations

import re
from collections.abc import Mapping
from dataclasses import dataclass, field
from enum import Enum
from hashlib import sha256
from typing import Any


class Stability(str, Enum):
    STATIC   = "static"
    SESSION  = "session"
    BRANCH   = "branch"
    TURN     = "turn"
    VOLATILE = "volatile"


class BlockKind(str, Enum):
    TOOL_SCHEMA   = "tool_schema"
    SYSTEM        = "system"
    CODING_POLICY = "coding_policy"
    REPO_SUMMARY  = "repo_summary"
    REASONBLOCK   = "reasonblock"
    FILE_SUMMARY  = "file_summary"
    USER_TASK     = "user_task"
    GIT_DIFF      = "git_diff"
    TOOL_RESULT   = "tool_result"
    SCRATCHPAD    = "scratchpad"


DEFAULT_STABILITY: dict[BlockKind, Stability] = {
    BlockKind.TOOL_SCHEMA:   Stability.STATIC,
    BlockKind.SYSTEM:        Stability.STATIC,
    BlockKind.CODING_POLICY: Stability.STATIC,
    BlockKind.REPO_SUMMARY:  Stability.SESSION,
    BlockKind.REASONBLOCK:   Stability.BRANCH,
    BlockKind.FILE_SUMMARY:  Stability.BRANCH,
    BlockKind.USER_TASK:     Stability.TURN,
    BlockKind.GIT_DIFF:      Stability.TURN,
    BlockKind.TOOL_RESULT:   Stability.TURN,
    BlockKind.SCRATCHPAD:    Stability.VOLATILE,
}

_NON_CACHEABLE = {Stability.TURN, Stability.VOLATILE}
_ID_RE = re.compile(r"^[a-z0-9_./:-]+$")


@dataclass(frozen=True)
class PromptBlock:
    id: str
    kind: BlockKind
    content: str
    stability: Stability
    cacheable: bool = True
    metadata: Mapping[str, Any] = field(default_factory=dict)
    stability_override_reason: str | None = None

    def __post_init__(self) -> None:
        if not self.content:
            raise ValueError("PromptBlock.content must not be empty")
        if not _ID_RE.match(self.id):
            raise ValueError(
                f"PromptBlock.id {self.id!r} is invalid; must match ^[a-z0-9_./:-]+$"
            )
        default_stab = DEFAULT_STABILITY[self.kind]
        if self.stability != default_stab and not self.stability_override_reason:
            raise ValueError(
                f"PromptBlock stability {self.stability!r} overrides the default "
                f"{default_stab!r} for kind {self.kind!r}; provide stability_override_reason"
            )
        # Force cacheable=False for TURN and VOLATILE regardless of caller intent.
        if self.stability in _NON_CACHEABLE and self.cacheable:
            object.__setattr__(self, "cacheable", False)

    @property
    def version_hash(self) -> str:
        return sha256(self.content.encode("utf-8")).hexdigest()

    @property
    def token_estimate(self) -> int:
        from .tokens import estimate_tokens
        return estimate_tokens(self.content)
```

Notes:
- Use `@property` (not `cached_property`) because `frozen=True` dataclass has no `__dict__` for
  caching — `sha256` is fast enough that recomputation is acceptable; if profiling shows cost,
  a caller can cache the block reference instead.
- `object.__setattr__` is needed to mutate a `frozen=True` field inside `__post_init__`.

```python
"""Token estimation for prompt blocks.

Primary: tiktoken cl100k_base (matches repo_map/budget.py pattern).
Fallback: max(1, len(text) // 4) when tiktoken is unavailable.
"""
from __future__ import annotations

import logging

logger = logging.getLogger(__name__)

_encoding: "tiktoken.Encoding | None" = None
_tiktoken_available: bool | None = None


def _get_encoding() -> "tiktoken.Encoding | None":
    global _encoding, _tiktoken_available
    if _tiktoken_available is None:
        try:
            import tiktoken
            _encoding = tiktoken.get_encoding("cl100k_base")
            _tiktoken_available = True
        except Exception:
            _tiktoken_available = False
            logger.debug("tiktoken unavailable; using char/4 fallback for token estimation")
    return _encoding


def estimate_tokens(text: str, model: str | None = None) -> int:
    """Estimate token count. Uses tiktoken when available, char/4 fallback otherwise."""
    enc = _get_encoding()
    if enc is not None:
        return len(enc.encode(text))
    return max(1, len(text) // 4)
```

Note: `model` parameter is accepted for forward compatibility (P3 may add provider-specific
tokenizers) but is ignored in this iteration; tiktoken's cl100k_base is used regardless.

"""Unit tests for PromptBlock, Stability, BlockKind, DEFAULT_STABILITY."""
import pytest
from atelier.core.capabilities.prompt_compilation import (
    BlockKind,
    DEFAULT_STABILITY,
    PromptBlock,
    Stability,
)


def _block(kind: BlockKind, **kwargs) -> PromptBlock:
    defaults = {"id": "test:block", "content": "hello world", "stability": DEFAULT_STABILITY[kind]}
    defaults.update(kwargs)
    return PromptBlock(kind=kind, **defaults)


def test_default_stability_per_kind() -> None:
    for kind in BlockKind:
        block = _block(kind)
        assert block.stability == DEFAULT_STABILITY[kind]


def test_override_requires_reason() -> None:
    with pytest.raises(ValueError, match="stability_override_reason"):
        PromptBlock(
            id="test:block",
            kind=BlockKind.SYSTEM,
            content="system prompt",
            stability=Stability.BRANCH,  # overrides STATIC default
        )


def test_override_with_reason_accepted() -> None:
    block = PromptBlock(
        id="test:block",
        kind=BlockKind.SYSTEM,
        content="system prompt",
        stability=Stability.BRANCH,
        stability_override_reason="per-session system prompt variant",
    )
    assert block.stability == Stability.BRANCH


def test_version_hash_stable_across_processes() -> None:
    block = _block(BlockKind.SYSTEM, content="hello world")
    known = "b94d27b9934d3e08a52e52d7da7dabfac484efe04294e576e86e0b359c17d9e"
    # sha256("hello world") known value (first 63 chars shown for readability)
    assert block.version_hash == "b94d27b9934d3e08a52e52d7da7dabfac484efe04294e576e86e0b359c17d9e09"[:len(block.version_hash)]
    # More robustly: check it's a valid 64-char hex string and stable
    import hashlib
    assert block.version_hash == hashlib.sha256("hello world".encode()).hexdigest()


def test_volatile_forces_uncacheable() -> None:
    block = PromptBlock(
        id="test:block",
        kind=BlockKind.SCRATCHPAD,
        content="scratchpad content",
        stability=Stability.VOLATILE,
        cacheable=True,  # caller intent overridden
    )
    assert block.cacheable is False


def test_turn_forces_uncacheable() -> None:
    block = _block(BlockKind.USER_TASK)
    assert block.cacheable is False


def test_empty_content_raises() -> None:
    with pytest.raises(ValueError, match="must not be empty"):
        PromptBlock(id="test:block", kind=BlockKind.SYSTEM, content="", stability=Stability.STATIC)


def test_invalid_id_raises() -> None:
    with pytest.raises(ValueError, match="invalid"):
        PromptBlock(id="Test Block!", kind=BlockKind.SYSTEM, content="x", stability=Stability.STATIC)


def test_static_block_cacheable_by_default() -> None:
    block = _block(BlockKind.TOOL_SCHEMA)
    assert block.cacheable is True


def test_all_block_kinds_have_default_stability() -> None:
    assert set(DEFAULT_STABILITY.keys()) == set(BlockKind)
```

"""Unit tests for the token estimator."""
import pytest
from atelier.core.capabilities.prompt_compilation.tokens import estimate_tokens


def test_tiktoken_used_when_available() -> None:
    # If tiktoken is installed, result should be > 0 and precise
    result = estimate_tokens("Hello, world!")
    assert isinstance(result, int)
    assert result > 0


def test_returns_positive_integer_for_short_text() -> None:
    assert estimate_tokens("x") >= 1


def test_longer_text_gives_larger_estimate() -> None:
    short = estimate_tokens("Hi")
    long_text = "This is a much longer paragraph that should result in more tokens than a short greeting."
    long = estimate_tokens(long_text)
    assert long > short


def test_char_fallback_within_15_percent() -> None:
    """The char/4 fallback should be within 15% of tiktoken for prose."""
    text = (
        "The quick brown fox jumps over the lazy dog. "
        "Pack my box with five dozen liquor jugs. "
        "How razorback-jumping frogs can level six piqued gymnasts."
    )
    tiktoken_estimate = estimate_tokens(text)
    char_fallback = max(1, len(text) // 4)
    # Allow 15% drift either way
    assert abs(tiktoken_estimate - char_fallback) / max(tiktoken_estimate, 1) < 0.15


def test_empty_string_returns_at_least_one() -> None:
    # estimate_tokens("") should not crash; fallback: max(1, 0//4) = 1
    # tiktoken: empty string encodes to [] so len([]) = 0; we handle below
    result = estimate_tokens("")
    assert result >= 0  # tiktoken returns 0 for empty; fallback returns 1
```

## Risks

| Risk | Mitigation |
|------|-----------|
| `frozen=True` + `__post_init__` mutation via `object.__setattr__` may surprise mypy | Use `# type: ignore[misc]` with a comment explaining the intent |
| `cached_property` incompatible with `frozen=True` + `slots=True` | Use `@property` (no caching) as specified |
| tiktoken encoding lazy-init is not thread-safe without a lock | Module-level `_get_encoding()` with a boolean flag is fine for single-threaded CLI; a lock can be added in P5 if needed |
