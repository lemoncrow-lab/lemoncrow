import { Component, type ErrorInfo, type ReactNode } from "react";

interface ErrorBoundaryProps {
  children: ReactNode;
  /** Custom fallback renderer. Receives the error and a reset callback. */
  fallback?: (error: Error, reset: () => void) => ReactNode;
  /** Hook for external logging (Sentry, Datadog, etc.). */
  onError?: (error: Error, info: ErrorInfo) => void;
}

interface ErrorBoundaryState {
  error: Error | null;
}

export class ErrorBoundary extends Component<
  ErrorBoundaryProps,
  ErrorBoundaryState
> {
  state: ErrorBoundaryState = { error: null };

  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { error };
  }

  componentDidCatch(error: Error, info: ErrorInfo): void {
    // Log full details for debugging.
    console.error("ErrorBoundary caught an error:", error, info.componentStack);
    this.props.onError?.(error, info);
  }

  reset = (): void => {
    this.setState({ error: null });
  };

  render(): ReactNode {
    const { error } = this.state;

    if (error) {
      if (this.props.fallback) {
        return this.props.fallback(error, this.reset);
      }

      return (
        <div role="alert" style={styles.container}>
          <h2 style={styles.title}>Something went wrong</h2>
          <pre style={styles.message}>{error.message}</pre>
          <button type="button" onClick={this.reset} style={styles.button}>
            Retry
          </button>
        </div>
      );
    }

    return this.props.children;
  }
}

const styles = {
  container: {
    padding: "1.5rem",
    border: "1px solid #f5c2c7",
    borderRadius: "0.5rem",

    color: "#842029",
    maxWidth: "32rem",

  },

Key points:
- Must be class — error boundaries need `getDerivedStateFromError` + `componentDidCatch`, no hook version.
- `getDerivedStateFromError` sets fallback state. `componentDidCatch` logs (console + optional `onError` hook).
- Retry button calls `reset`, clears error, re-renders `children`.
- Optional `fallback` prop for custom UI.

Usage:

```tsx
<ErrorBoundary onError={(e, info) => sendToSentry(e, info)}>
  <App />
</ErrorBoundary>
```

Gotcha: only catches errors in render/lifecycle/constructor of children. Not event handlers, async, or SSR — use try/catch there. Want plain JS version, or reset that also re-runs a failed data fetch?
