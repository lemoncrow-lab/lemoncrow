  element: NonDeletedExcalidrawElement,
  elementsMap: NonDeletedSceneElementsMap,
  renderConfig: StaticCanvasRenderConfig,
  appState: StaticCanvasAppState | InteractiveCanvasAppState,

  const zoom: Zoom = renderConfig

    : {
        value: 1 as NormalizedZoomValue,
      };
  const prevElementWithCanvas = elementWithCanvasCache.get(element);
  const shouldRegenerateBecauseZoom =
    prevElementWithCanvas &&
    prevElementWithCanvas.zoomValue !== zoom.value &&
    !appState?.shouldCacheIgnoreZoom;
  const boundTextElement = getBoundTextElement(element, elementsMap);
  const boundTextElementVersion = boundTextElement?.version || null;
  const imageCrop = isImageElement(element) ? element.crop : null;

  const containingFrameOpacity =
    getContainingFrame(element, elementsMap)?.opacity || 100;

  if (
    !prevElementWithCanvas ||
    shouldRegenerateBecauseZoom ||
    prevElementWithCanvas.theme !== appState.theme ||
    prevElementWithCanvas.boundTextElementVersion !== boundTextElementVersion ||
    prevElementWithCanvas.imageCrop !== imageCrop ||
    prevElementWithCanvas.containingFrameOpacity !== containingFrameOpacity ||
    // since we rotate the canvas when copying from cached canvas, we don't
    // regenerate the cached canvas. But we need to in case of labels which are
    // cached alongside the arrow, and we want the labels to remain unrotated
    // with respect to the arrow.
    (isArrowElement(element) &&
      boundTextElement &&
      element.angle !== prevElementWithCanvas.angle)
  ) {
    const elementWithCanvas = generateElementCanvas(
      element,
      elementsMap,
      zoom,
      renderConfig,
      appState,
    );

    if (!elementWithCanvas) {
      return null;
    }

    elementWithCanvasCache.set(element, elementWithCanvas);

    return elementWithCanvas;
  }
  return prevElementWithCanvas;
};

[assistant] ## Rendering model

Excalidraw uses **two stacked `<canvas>` layers**, each a `React.memo` component that redraws only when its inputs change:

- **`StaticCanvas`** (`components/canvases/StaticCanvas.tsx`) — the elements themselves. Calls `renderStaticScene`.
- **`InteractiveCanvas`** — selection boxes, handles, remote cursors, snap lines. Calls `renderInteractiveScene`.

Splitting them means moving the selection outline never forces every shape to repaint.

## Update pipeline (element → pixels)

**1. Scene → renderable set** (`scene/Renderer.ts`)
`Renderer.getRenderableElements` pulls non-deleted elements from `Scene`, builds a `RenderableElementsMap` (drops the currently-edited text and the new-element-in-frame), then filters to **viewport-visible** elements via `isElementInViewport`. This whole step is `memoize`d on a `canvasNonce` = `scene.getSceneNonce()` (+ new-element versionNonce). Scene bumps `sceneNonce = randomInteger()` on any mutation (`Scene.ts:304`) → nonce change is the top-level "something changed, recompute" signal.

**2. React memo gate** (`StaticCanvas.areEqual`)
Re-renders only if `canvasNonce`, `scale`, the `elementsMap`/`visibleElements` refs, relevant `appState` props (`isShallowEqual` over an explicit whitelist), or `renderConfig` changed. Otherwise skips entirely.

**3. Scene draw** (`renderer/staticScene.ts` → `_renderStaticScene`)
Clears canvas, applies zoom/scroll transform, then `visibleElements.forEach(renderElement)`. Wrapped in `renderStaticSceneThrottled = throttleRAF(...)` — when throttling is on, paints at most once per animation frame (coalesces bursts).

**4. Per-element draw + caches** (`element/renderElement.ts`)
Each element is rendered from an **offscreen per-element canvas**, not redrawn from scratch:
- `elementWithCanvasCache` — `WeakMap<element, canvas>`. `generateElementWithCanvas` regenerates the element's bitmap only when zoom, theme, bound-text version, image crop, containing-frame opacity, or (arrow-label) angle changed; else returns the cached canvas. `drawElementFromCanvas` then just blits + rotates it.
- `ShapeCache` (`shape.ts`) — `WeakMap<element, roughjs drawable>`. Caches the rough.js shape generation one layer below.

So a redraw is: nonce invalidates the memoized renderable set → React memo lets the changed canvas through → RAF-throttled scene loop → each element blitted from its WeakMap-cached bitmap, regenerating only the few whose visual inputs actually changed. `freedraw`/export paths bypass the cache and draw directly.

Key files: `scene/Renderer.ts`, `scene/Scene.ts` (nonce), `renderer/staticScene.ts`, `renderer/interactiveScene.ts`, `element/renderElement.ts`, `element/shape.ts`, `components/canvases/*`.
